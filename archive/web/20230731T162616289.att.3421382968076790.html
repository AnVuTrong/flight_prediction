def get_data():
    train_data_path = './datafiles/Data-Route-6-280-flights-21.5.23.xlsx'
    train = pd.read_excel(train_data_path)
    train =train.drop(train[train.Ac_feet < 5487].index)

    train = train.replace(regex=['', '','','',' ',' ',' ',' ',' '], value='')
    train=train.fillna(0)
    train =train.replace(0, 0.0005)
    train.drop(['Ac_id','Ac_code'], inplace=True, axis=1)
    for typ in train.columns.values:
        if typ != 'Ac_id':
            train[typ] = pd.to_numeric(train[typ], downcast="float")         
    target1 = train.Ac_Lat
    target2 = train.Ac_Lon
    target3 = train.Ac_feet
    train.drop(['Ac_Lat','Ac_Lon','Ac_feet'], inplace=True, axis=1)
    return train, target1,target2,target3
def ANN_MODEL(lr):
  inp= Input(shape = (41,))
  x = Flatten()(inp)
  x1=(tf.keras.layers.Dense(256, input_dim=(41,),activation = 'relu'))(x)
  x2=(tf.keras.layers.Dense(512, activation = 'relu'))(x1)
  x3=(tf.keras.layers.Dense(512, activation = 'relu'))(x2)
  x4=(tf.keras.layers.Dense(256, activation = 'relu'))(x3)
  x5=(tf.keras.layers.Dense(128, activation = 'relu'))(x4)
  x6=(tf.keras.layers.Dense(64, activation = 'relu'))(x5)
  out1=(tf.keras.layers.Dense(1, activation='linear',kernel_regularizer=l2(0.01), bias_regularizer=l2(0.01)))(x6)
  model = Model(inp, outputs=out1)  
  model.compile(loss='mean_absolute_error', 
                optimizer=Adam(lr=lr))
  return model
